# Cara Mengambil Inputan dari user

print("----- MENGAMBIL DATA DARI USER -----")

# ini cara mengambil data dari user dengan type data String
data_str = str(input("--- Masukan data String : "))
# ini cara mengambil data dari user dengan type data Integer
data_int = int(input("--- Masukan data integer : "))
# ini cara mengambil data dari user dengan type data float
data_float = float(input("--- Masukan data float : "))
# ini cara mengambil data dari user dengan type data bool
data_bool = bool(input("--- Masukan data boolean : "))

# ini menampilkan data inputan user yang telah di input

print("----------------------------------------")
print("Data String : ",data_str,type(data_str))
print("Data integer : ",data_int,type(data_int))
print("Data float : ",data_float,type(data_float))
print("Data boolean : ",data_bool,type(data_bool))
print("----------------------------------------")
